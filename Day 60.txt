class laptop():
    chargertype="C-Type"
    
    def __init__(self):
        self.brand=""
        self.price=34
    
    def setPrice(self,price):
        self.price=price
    
    def getPrice(self):
        print(self.price)
        
    @classmethod
    def changechargertype(self):
        self.chargertype="B-Type"
        print("Charger type changed to B")
        
    @staticmethod
    def Info():
        print("This is laptop class")
        
    
hp=laptop()
hp.setPrice(20000)
hp.getPrice()

hp.changechargertype()
hp.Info()
